 3024:   List<CoaAccount> _buildTreeStructure() {
 3025:     // Create a map using uniqueId for duplicate-safe lookup
 3026:     final uidMap = <String, CoaAccount>{};
 3027:     final codeMap = <String, List<CoaAccount>>{};
 3028:     for (final acc in _accounts) {
 3029:       acc.children = [];
 3030:       uidMap[acc.uniqueId] = acc;
 3031:       codeMap.putIfAbsent(acc.code, () => []);
 3032:       codeMap[acc.code]!.add(acc);
 3033:     }
 3034: 
 3035:     // Build parent-child relationships using uniqueId when available
 3036:     final roots = <CoaAccount>[];
 3037:     for (final acc in _accounts) {
 3038:       bool attached = false;
 3039: 
 3040:       // Try uniqueId-based parent first (duplicate-aware)
 3041:       if (acc.parentUniqueId.isNotEmpty && uidMap.containsKey(acc.parentUniqueId)) {
 3042:         final parent = uidMap[acc.parentUniqueId]!;
 3043:         parent.children = [...parent.children, acc];
 3044:         attached = true;
 3045:       }
 3046:       // Fallback to code-based parent (non-duplicate case)
 3047:       else if (acc.parentCode.isNotEmpty && codeMap.containsKey(acc.parentCode)) {
 3048:         final candidates = codeMap[acc.parentCode]!;
 3049:         if (candidates.length == 1) {
 3050:           candidates.first.children = [...candidates.first.children, acc];
 3051:           attached = true;
 3052:         } else {
 3053:           // Multiple parents with same code â€” use name classification
 3054:           final myCategory = _classifyByName(acc.name);
 3055:           CoaAccount? bestParent;
 3056:           for (final p in candidates) {
 3057:             if (myCategory.isNotEmpty && _classifyByName(p.name) == myCategory) {
 3058:               bestParent = p;
 3059:               break;
 3060:             }
 3061:           }
 3062:           bestParent ??= candidates.first;
 3063:           bestParent.children = [...bestParent.children, acc];
 3064:           attached = true;
 3065:         }
 3066:       }
 3067: 
 3068:       if (!attached) {
 3069:         roots.add(acc);
 3070:       }
 3071:     }
 3072: 
 3073:     return roots;
 3074:   }
